입력받은 문자열을 각 단어로 쪼개기

import fs = require('fs');
const inputString = fs.readFileSync('/dev/stdin').toString();
const inputLineArray = inputString.split('\n');
let input = [];
for(const inputLine of inputLineArray){
    const wordArray = inputLine.split(' ');
    input.push(wordArray);
};

입력받은 문자열을 각 숫자로 쪼개기

import fs = require('fs');
const inputString = fs.readFileSync('/dev/stdin').toString();
const inputLineArray = inputString.split('\n');
let input = [];
for(const inputLine of inputLineArray){
    const wordArray = inputLine.split(' ');
    let numberArray = [];
    for(const word of wordArray){
        numberArray.push(Number(word));
    };
    input.push(numberArray);
};
